view FalsityNegTruth : ?Falsity → ?NegationTruth =
  include ?Propositions ❙

  include ?FalsityNegTruth❙
  include ?Proofs❙
  falseE = [p] p notE trueI ❙

  include ?Propositions ❙

  include ?Propositions ❙ 
  include ?Proofs❙
  include ?TruthNegFalsity❙
